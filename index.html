<script>
    // 自動設定今天日期
    document.getElementById('date').value = new Date().toISOString().split('T')[0];

    function generateDoc() {
        const getCheck = (id) => document.getElementById(id).checked ? "■" : "□";
        const val = (id) => document.getElementById(id).value;

        // 計算金額
        const q1 = parseFloat(val('qty1')) || 0;
        const p1 = parseFloat(val('price1')) || 0;
        const sub1 = q1 * p1;
        const tax = Math.round(sub1 * 0.05);

        const data = {
            vender: val('vender'), date: val('date'), order_no: val('order_no'),
            tel: val('tel'), contact: val('contact'), buyer: val('buyer'),
            // 類別勾選 
            c_1: getCheck('c_1'), c_2: getCheck('c_2'), c_3: getCheck('c_3'), c_4: getCheck('c_4'), c_5: getCheck('c_5'),
            // 付款方式勾選 
            p_1: getCheck('p_1'), p_2: getCheck('p_2'), p_3: getCheck('p_3'), p_4: getCheck('p_4'), 
            p_5: getCheck('p_5'), p_6: getCheck('p_6'), p_7: getCheck('p_7'), p_8: getCheck('p_8'),
            // 表格資料 
            item1: val('item1'), qty1: q1, price1: p1.toLocaleString(), subtotal1: sub1.toLocaleString(),
            total_net: sub1.toLocaleString(), tax: tax.toLocaleString(), grand_total: (sub1 + tax).toLocaleString()
        };

        JSZipUtils.getBinaryContent("template.docx", function(error, content) {
            if (error) { alert("找不到 template.docx 檔案！"); return; }
            const zip = new PizZip(content);
            const doc = new window.docxtemplater(zip, { paragraphLoop: true, linebreaks: true });
            doc.render(data);
            const out = doc.getZip().generate({
                type: "blob",
                mimeType: "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
            });
            saveAs(out, `訂購單_${val('vender')}.docx`);
        });
    }
</script>
